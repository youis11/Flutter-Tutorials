Circle Shape

Containers with a circular shape

                class Bola extends StatelessWidget {
                    final Color color;
                    Bola(this.color);

                    @override
                    Widget build(BuildContext context) {
                        return Expanded(
                        child: AspectRatio(
                            aspectRatio: 1,
                            child: Container(
                            decoration: BoxDecoration(
                                shape: BoxShape.circle,
                                color: this.color,
                            ),
                            ),
                        ),
                        );
                    }
                }
