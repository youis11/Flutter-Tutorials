Alert Dialog

PopUp actioned in AppBar that erases a list

                //ACTIONS IN APPBAR----------------------------------

                appBar: AppBar(
                    actions: [
                    IconButton(
                        icon: Icon(Icons.delete),
                        onPressed: _maybeEraseChecked,
                    )
                    ],
                ),

                //ALERT DIALOG---------------------------------------
                void _maybeEraseChecked() {
                    showDialog(
                    context: context,
                    builder: (innerContext) => AlertDialog(
                        title: Text('Confirmation'),
                        content: Text(
                            'Are you sure you want to delete the checked items?',
                        ),
                        actions: [
                            FlatButton(
                                onPressed: () {
                                    Navigator.of(innerContext).pop(false);
                                },
                                child: Text('Cancel'),
                            ),
                            FlatButton(
                                onPressed: () {
                                    Navigator.of(innerContext).pop(true);
                                },
                                child: Text('Erase'),
                            )
                        ],
                    ),
                    ).then((erase) {
                        if (erase) {
                            _eraseChecked();
                        }
                    });
                }

                void _eraseChecked() {
                    List pending = [];
                    for (var todo in _todos) {
                        if (!todo.done) pending.add(todo);
                    }
                    setState(() {
                        _todos = pending;
                    });
                }
