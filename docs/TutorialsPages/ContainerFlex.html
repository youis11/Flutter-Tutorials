Container Flex

Containers that have different flex values

                class Banda extends StatelessWidget {
                    final int flex;
                    final String text;
                    final Color color;
                    Banda(this.flex, this.text, this.color);

                    @override
                    Widget build(BuildContext context) {
                        return Expanded(
                        flex: this.flex,
                            child: Container(
                                color: this.color,
                                child: Center(
                                    child: Text(
                                        this.text,
                                        style: TextStyle(
                                        fontSize: 24,
                                        color: Color.fromARGB(129, 255, 255, 255),
                                        ),
                                    ),
                                ),
                            ),
                        );
                    }
                }
