Collapsible text

Stateful Widget that collapses text with a Flat Button

                //PUSH NAVIGATOR-----------------------------------------------------------
                
                RaisedButton(
                    child: Text('Editar'),
                    onPressed: () {
                        Navigator.of(context).push(
                            MaterialPageRoute(
                                builder: (_) => EditaNumero(numero), //number that changes
                            ),
                        ).then((enter) {
                            if (enter != null) {
                                setState(() {
                                    numero = enter;
                                });
                            }
                        });
                    },
                ),

                //POP NAVIGATOR------------------------------------------------------------

                RaisedButton(
                    child: Text('Guardar'),
                    onPressed: () {
                        final int enter = int.parse(_controller.text);
                        Navigator.of(context).pop(enter);
                    },
                )
