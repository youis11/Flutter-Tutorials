                    children: [
                        for (int i = 0; i < 50; i++)

                        )
                    ],
                ),

                //GRID VIEW BUILDER---------------------------------------

                        crossAxisCount: 5,
                    ),
                    itemCount: 1000,
                    itemBuilder: (context, index) {
                        print(index);
                        return Card(
                            child: Center(
                                child: Text(
                                    '${index + 1}',
                                ),
